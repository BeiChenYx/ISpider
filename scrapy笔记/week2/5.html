5. 腾讯动漫项目爬虫实战

代码如下：

# -*- coding: utf-8 -*"-

from selenium import webdriver
import time
import re
import urllib.request
from selenium.webdriver.common.desired_capabilities import DesiredCapabilities

#创建一个浏览器，并做初始化设置
dcap = dict(DesiredCapabilities.PHANTOMJS)
dcap["phantomjs.page.settings.userAgent"]="Mozilla/5.0(WindowsNT6.1;rv:2.0.1)Gecko/20100101Firefox/4.0.1"
browser=webdriver.PhantomJS(desired_capabilities=dcap)
#brower=webdriver.phantomJS()

#
browser.get("http://ac.qq.com/ComicView/index/id/539443/cid/1?fromPrev=1")
#a=browser.get_screenshot_as_file("./dongman/test.png")

for i in range(10):
    #browser.execute_script('window.scrollTo(0,document.body.scrollHeight)')
    js='window.scrollTo('+str(i*1280)+','+str((i+1)*1280)+')'
    browser.execute_script(js)
    time.sleep(1)

#js="document.body.scrollTop=10000" #滚动条下拉1000px
#browser.execute_script(js)
#browser.implicitly_wait(300)
print(browser.current_url)
data=browser.page_source #用于获取页面html源代码
#网页另存为
fh=open("./dongman/dongman.html","w",encoding="utf-8")
fh.write(data)
fh.close()
browser.quit()

#爬取网页中的图片
pat='<img src="(http:\/\/ac\.tc\.qq\.com\/store_file_download\?.*?;name=.*?\.jpg)"'
allid=re.compile(pat).findall(data)
for i in range(0,len(allid)):
    thisurl=allid[i].replace("amp;","")
    print(thisurl)
    localpath="./dongman/"+str(i)+".jpg"
    urllib.request.urlretrieve(thisurl,filename=localpath)
