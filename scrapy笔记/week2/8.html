6. 分布式爬虫的实战

# 还原一个镜像
$ docker load --input mycrawl.tar

$ docker images

# 压缩保存一个镜像文件
$ docker save -o /mytest.tar c3e8

#创建运行一个镜像(中心节点，这样好支持远程连接)
$ docker run  -tid --name center 3a54

#进入中心节点服务容器
$ docker attach 2eaf

#查看网络连接
$ cat /etc/hosts

#测试mysql数据
$ /etc/init.d/mysql restart
$ mysql -h 127.0.0.1 -u root -p 

# 修改配置文件(支持远程连接)
$ vim /etc/mysql/my.cnf

    #注释掉此行
    bin-address = 127.0.0.1

# 重启mysql数据库后，再次测试就可以了
$ /etc/init.d/mysql restart
$ mysql -h 127.0.0.1 -u root -p
